// Função para salvar uma lista em uma planilha do Google Sheets
function salvarListaNoSheet(lista, sheet, startColumn) {
  var sheetData = [];
  
  for (var key in lista) {
    var rowData = [];
    for (var subKey in lista[key]) {
      rowData.push(lista[key][subKey]);
    }
    sheetData.push(rowData);
  }

  var range = sheet.getRange(startColumn + '1:' + String.fromCharCode(startColumn.charCodeAt(0) + Object.keys(lista).length - 1) + '1');
  range.setValues([Object.keys(lista)]);
  
  var dataRange = sheet.getRange(startColumn + '2:' + String.fromCharCode(startColumn.charCodeAt(0) + Object.keys(lista).length - 1) + (sheetData.length + 1));
  dataRange.setValues(sheetData);
}

function salvarImagensDrive(pdcaData, codigo){

  var planilha = SpreadsheetApp.openById(url_salvar);
  var sheetImagens = planilha.getSheetByName("IMAGEM");

  const email = Object.keys(pdcaData)[0];
  const pdca = pdcaData[email].pdca;

      // Salvar imagens na aba IMAGEM
    const quadros = ['PLAN'];
    const colunasImagens = ['P'];
    
    let todasImagens = [];
    
    quadros.forEach((quadro, index) => {
      const prefixoColuna = colunasImagens[index];
      const imagens = pdca[quadro].images;
      
      if (imagens.length > 0) {
        imagens.forEach(img => {
          // Salvar cada imagem no Drive e obter URL
          const imageUrl = salvarImagemNoDrive(img.url, quadro, codigo);
          
          if (imageUrl) {
            todasImagens.push([
              codigo,
              prefixoColuna,
              img.id,
              imageUrl  // URL da imagem no Drive
            ]);
          }
        });
      }
    });
    
    // Adicionar todas as imagens de uma vez (mais eficiente)
    if (todasImagens.length > 0) {
      sheetImagens.getRange(
        sheetImagens.getLastRow() + 1, 
        1, 
        todasImagens.length, 
        4
      ).setValues(todasImagens);
    }

}
